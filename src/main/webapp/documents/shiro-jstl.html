标签名称 标签条件（均是显示标签内容）
< shiro:authenticated> < / shiro:authenticated> 登录之后
< shiro:notAuthenticated> < /shiro:notAuthenticated> 不在登录状态时
< shiro:guest> < /shiro:guest> 用户在没有RememberMe时
< shiro:user> < /shiro:user> 用户在RememberMe时
< shiro:hasAnyRoles name="abc,123" > < /shiro:hasAnyRoles name="abc,123" > 在有abc或者123角色时
< shiro:hasRole name="abc"> < /shiro:hasRole name="abc"> 拥有角色abc
< shiro:lacksRole name="abc"> < /shiro:lacksRole name="abc"> 没有角色abc
< shiro:hasPermission name="abc"> < /shiro:hasPermission name="abc"> 拥有权限资源abc
< shiro:lacksPermission name="abc"> < /shiro:lacksPermission name="abc"> 没有abc权限资源
< shiro:principal> < /shiro:principal> 显示用户身份名称
< shiro:principal property="username"/> < /shiro:principal property="username"/>显示用户身份中的属性值
